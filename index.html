២. វិធីសាស្រ្តដាក់ឈ្មោះ BEM

CSS
/* BLOCK: ធាតុឯករាជ្យ */ .card { background: white; border-radius: 16px; } /* ELEMENT: ផ្នែកមួយនៃ block */ .card__head { background: linear-gradient(100deg, #0a1a3a, #122855); } .card__body { padding: 24px; } /* MODIFIER: កំណែប្រែនៃ block */ .btn--primary { background: linear-gradient(135deg, #f59e0b, #e68a00); } .btn--outline { background: transparent; border: 2px solid white; } .btn--blue { background: linear-gradient(135deg, #2563eb, #0a1a3a); }
BEM = Block + Element + Modifier:
  • Block — .card, .btn, .nav (ធាតុឯករាជ្យ)
  • Element — .card__head, .card__body (ផ្នែកនៃ block គូសបន្ទាត់ពីរដង)
  • Modifier — .btn--primary, .btn--outline (កំណែប្រែ គូសបន្ទាត់ពីរដង)
  • អត្ថប្រយោជន៍ — ការដាក់ឈ្មោះច្បាស់លាស់ គ្មានសង្គ្រាម specificity ងាយថែទាំ

/* ស្លាយនីមួយៗទទួលអានីមេហ្សង់ដូចគ្នាប៉ុន្តែការពន្យារពេលខុសគ្នា */ .hero__slide { position: absolute; inset: 0; opacity: 0; animation: heroSlide 15s infinite; } .hero__slide:nth-child(1) { animation-delay: 0s; } .hero__slide:nth-child(2) { animation-delay: 5s; } .hero__slide:nth-child(3) { animation-delay: 10s; }

  • ៥-១០s — ស្លាយ ២ មើលឃើញ ស្លាយ ១ និង ៣ លាក់
  • ១០-១៥s — ស្លាយ ៣ មើលឃើញ ស្លាយ ១ និង ២ លាក់
  • ១៥s — វដ្តរង្វិលជារៀងរហូត

២. វិធីសាស្រ្ត Checkbox Hack (ម៉ឺនុយទូរស័ព្ទ)

ម៉ឺនុយ hamburger ទូរស័ព្ទដំណើរការដោយគ្មាន JavaScript ណាមួយដោយប្រើប្រអប់ធីកលាក់និង pseudo-class :checked របស់ CSS។

HTML + CSS
<!-- HTML: ប្រអប់ធីកលាក់ + ស្លាក label មើលឃើញ --> <input type="checkbox" class="nav__checkbox" id="navToggle"> <label for="navToggle" class="nav__toggle">☰ ម៉ឺនុយ</label> <ul class="nav__menu">...</ul> /* CSS: លាក់ប្រអប់ធីក បង្ហាញម៉ឺនុយនៅពេលធីក */ .nav__checkbox { display: none; } .nav__menu { display: none; } .nav__checkbox:checked ~ .nav__menu { display: flex; }
របៀបដំណើរការ:
  • input[type="checkbox"] — លាក់ពីទិដ្ឋភាព (display: none)
  • label[for="navToggle"] — ចុចនេះប្តូរស្ថានភាពប្រអប់ធីក
  • ~ (general sibling selector) — ជ្រើសរើស .nav__menu ដែលនៅក្រោយប្រអប់ធីក
  • :checked — អនុវត្តតែនៅពេលប្រអប់ធីកបានធីក (ម៉ឺនុយបើក)
